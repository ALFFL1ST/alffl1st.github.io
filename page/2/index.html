GitHub教育福利申请完可以免费使用copilot中的各种模型，涵盖cloude，gpt和Gemini的主流付费模型，所以申请成功之后还是很超值的

跟随步骤一步一步做能申请下来

1.设置支付信息

信息差不多这么设置，得全部使用英文，中文名字的话第一格写姓第二格写名实测可以

学校名也写英文

到这个界面会调用摄像头，手持手机的一个信息表格截图就能过

就是建一个这样的表格，把基础信息用英文填完，最下层的time需要跟申请当天的信息一致，其他信息与支付信息保持一致，手持可以不露脸

记得最好不要挂梯子，不过挂梯子的验证方式也有，会麻烦一些

DWORD ReadPEFile(/*LPSTR lpszFile,*/ LPVOID* pFileBuffer) {
	FILE* pFile = NULL;
	DWORD fileSize = 0;
	LPVOID ptempFileBuffer = NULL;
	//打开文件
	pFile = fopen(/*lpszFile*//*"c:\\Users\\10794\\Desktop\\PETool 1.0.0.5.exe"*/"c:\\Users\\10794\\Desktop\\notepad.exe", "rb");
	if (!pFile) {
		printf("打开文件失败\n");
		return 0;
	}
	//读取文件大小
	fseek(pFile, 0, SEEK_END);
	fileSize = ftell(pFile);
	fseek(pFile, 0, SEEK_SET);
	//分配空间
	ptempFileBuffer = malloc(fileSize);
	if (!ptempFileBuffer) {
		printf("分配空间失败\n");
		fclose(pFile);
		return 0;
	}
	//将文件读入缓冲区
	size_t n = fread(ptempFileBuffer, fileSize, 1, pFile);
	if (!n) {
		printf("读取文件失败\n");
		free(ptempFileBuffer);
		fclose(pFile);
		return 0;
	}
	*pFileBuffer = ptempFileBuffer;
	ptempFileBuffer = NULL;
	printf("读取成功\n");
	fclose(pFile);
	return fileSize;
}

{
	LPVOID pFileBuffer = NULL;
	PIMAGE_DOS_HEADER pDos_header = NULL;
	PIMAGE_NT_HEADERS pNT_header = NULL;
	PIMAGE_FILE_HEADER pPE_header = NULL;
	PIMAGE_OPTIONAL_HEADER pOption_header = NULL;
	PIMAGE_SECTION_HEADER pSection_header = NULL;

	//读取文件函数读取
	pFileBuffer = Read_PE();

	//验证MZ标志
	pDos_header = (PIMAGE_DOS_HEADER)pFileBuffer;
	if (pDos_header->e_magic != IMAGE_DOS_SIGNATURE)
	{
		printf("不是有效的MZ标志\n");
		free(pFileBuffer);
	}
	//打印DOS头
	printf("DOS头信息\n");
	printf("PE文件偏移%x\n", pDos_header->e_lfanew);

	//计算NT头起始地址*
	pNT_header = (PIMAGE_NT_HEADERS)((DWORD)pFileBuffer + pDos_header->e_lfanew);

	//验证NT头
	if (pNT_header->Signature != IMAGE_NT_SIGNATURE)
	{
		printf("不是有效的PE标志\n");
		free(pFileBuffer);
		return;
	}

	//打印标准PE头和可选PE头
	pPE_header = (PIMAGE_FILE_HEADER)((DWORD)pNT_header + 4);
	printf("标准PE头\n");
	printf("PE程序运行的CPU型号:%x\n", pPE_header->Machine);
	printf("节的数量：%x\n", pPE_header->NumberOfSections);
	printf("时间戳：%x\n", pPE_header->TimeDateStamp);
	pOption_header = (PIMAGE_OPTIONAL_HEADER)((DWORD)pPE_header + IMAGE_SIZEOF_FILE_HEADER);
	printf("打印可选PE头\n");
	printf("说明文件类型:%x\n", pOption_header->Magic);
	printf("内存镜像基址:%x\n", pOption_header->ImageBase);
	printf("程序入口:%x\n", pOption_header->AddressOfEntryPoint);
	pSection_header = (PIMAGE_SECTION_HEADER)((DWORD)pOption_header + pPE_header->SizeOfOptionalHeader);
	printf("打印节表\n");
	for (int i=0;i< pPE_header->NumberOfSections;i++, pSection_header++) {
		printf("name:%s\n", pSection_header->Name);
		printf("节区在内存中的偏移地址:%x\n", pSection_header->VirtualAddress);
		printf("节在文件中对齐后的尺寸:%x\n", pSection_header->SizeOfRawData);
		printf("节区在文件中的偏移地址:%x\n", pSection_header->PointerToRawData);
	}
	//释放内存
	free(pFileBuffer);
}

汇编基础指令集

发表于 2025-12-17
  1. 加载指令：将数据从内存加载到寄存器

MOV A, B ; 将寄存器B的值加载到寄存器A
2. 存储指令：将寄存器中的数据存储到内存

MOV B, A ; 将寄存器A的值存储到寄存器B
3. 加法指令：

ADD A, B, C ; 将寄存器B和寄存器C中的值相加，结果存储在寄存器A中
4. 减法指令：

SUB A, B, C ; 将寄存器B和寄存器C中的值相减，结果存储在寄存器A中
5. 乘法指令：

MUL A, B, C ; 将寄存器B和寄存器C中的值相乘，结果存储在寄存器A中
6. 除法指令：

DIV A, B, C ; 将寄存器B和寄存器C中的值相除，结果存储在寄存器A中
7. 比较指令：比较两个寄存器中的值，根据结果设置标志位

CMP A, B ; 比较寄存器A和寄存器B中的值，根据结果设置标志位
8. 跳转指令：根据标志位或者条件，跳转到指定的位置执行

JMP label ; 无条件跳转到label处执行

JE label ; 如果相等，跳转到label处执行

JNE label ; 如果不相等，跳转到label处执行
9. 堆栈操作指令：操作堆栈，如入栈、出栈等

PUSH A ; 将寄存器A的值压入堆栈

POP B ; 将堆栈栈顶的值弹出并存储到寄存器B中
10. HLT ; 停止指令，指示程序执行完毕

11.MOV EAX,[EBX]:指把寄存器EBX内的值作为当前相对于段寄存器DS的偏移地址，将该地址的值存入寄存器EAX中

12.INC EAX:指将EAX内的数据自增1

13.LOOP S:判断cx（计数寄存器的值），不为零则转至标号S处执行程序，如果为0则向下执行－－操作原理①CX寄存器内值自减②判断CX寄存器的值－－>常使用cx存储要循环的次数

14.代码段结束语句:

MOV EAX,4C00H

INT 21H

CODE ENDS

END

15.汇编程序中数据的开头不能为字母，若为字母要在前面补0

16.ASM文件中可使用DS:[N]来把相对段地址偏移地址为n的地址内的值传给寄存器（赋值操作）

17.0:200到0:2ff为安全空间，可以进行编辑

18.DW－－>define word—－>定义字型数据，定义的每个字型数据之间以逗号分隔，它们所占的内存空间大小为16字节（两个字节位）DB为一半

19.标号START决定了程序起始位置的代码行，在程序全部结束时要写成END START
